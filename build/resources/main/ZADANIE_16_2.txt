SELECT L.NAME, S.SUMMARY FROM ISSUESLISTS L
JOIN ISSUES S
ON L.ID = S.ISSUESLIST_ID;

SELECT S.SUMMARY, U.FIRSTNAME, U.LASTNAME FROM USERS U
JOIN ISSUES S
ON U.ID = S.USER_ID_ASSIGNEDTO;

SELECT S.SUMMARY, U.FIRSTNAME, U.LASTNAME, COUNT(U.ID) AS "Ilość zadań" FROM USERS U
JOIN ISSUES S
ON U.ID = S.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
